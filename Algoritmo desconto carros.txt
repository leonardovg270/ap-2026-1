Algoritmo <desconto carros>

Inicio
Real: valor, desconto, valor_final
Inteiro: ano, total_ate_2010, total_geral
Caracter: opcao

total_ate_2010 <- 0
total_geral <- 0

Enquanto (opcao != "N")

    escreva ("Digite o valor do carro: ")
    leia (valor)

    escreva ("Digite o ano do carro: ")
    leia (ano)

    se (ano <= 2010) então
        desconto <- valor * 0.12
        total_ate_2010 <- total_ate_2010 + 1
    senao
        desconto <- valor * 0.07
    fimse

    valor_final <- valor - desconto

    escreva ("Desconto: R$ " + desconto)
    escreva ("Valor final: R$ " + valor_final)

    total_geral <- total_geral + 1

    escreva ("Deseja continuar? (S/N): ")
    leia (opcao)

fimEnquanto

escreva ("Total de carros até 2010: " + total_ate_2010)
escreva ("Total geral de carros: " + total_geral)

Fim
